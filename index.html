JSON Extractor â€” Python Appendix A, Snippet 3
Python 
def extract_json(claude_response_text):
    text = claude_response_text
    start = text.find('{')
    end = text.rfind('}') + 1
    if start == -1 or end == 0:
        raise ValueError("No JSON object found in response")
    json_string = text[start:end]
    return json.loads(json_string)

# Usage:
raw_text = message.content[0].text
parsed = extract_json(raw_text)

Prospect CSV Researcher â€” Python Appendix A, Snippet 5
Python 
import anthropic
import csv
import json
import time

client = anthropic.Anthropic(api_key="your-api-key-here")  # CHANGE THIS

def research_prospect(company, why_relevant):
    response = client.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=400,
        messages=[{
            "role": "user",
            "content": f"Research this prospect and return JSON.\
\
Company: {company}\
Why we are reaching out: {why_relevant}\
\
Return ONLY: {{\
  'best_hook': 'one sentence',\
  'key_fact': 'one specific fact'\
}}"
        }]
    )
    text = response.content[0].text
    start = text.find('{')
    end = text.rfind('}') + 1
    return json.loads(text[start:end])

with open('prospects.csv', 'r') as infile, open('prospects_researched.csv', 'w', newline='') as outfile:
    reader = csv.DictReader(infile)
    fieldnames = reader.fieldnames + ['best_hook', 'key_fact']
    writer = csv.DictWriter(outfile, fieldnames=fieldnames)
    writer.writeheader()
    for row in reader:
        print(f"Processing: {row['Company']}")
        try:
            result = research_prospect(row['Company'], row['Why Relevant'])
            row.update(result)
            writer.writerow(row)
            time.sleep(1)  # Rate limiting - do not remove
        except Exception as e:
            print(f"Error on {row['Company']}: {e}")
            row['best_hook'] = 'ERROR'
            row['key_fact'] = str(e)
            writer.writerow(row)
Email Sanitizer â€” Python Appendix A, Snippet 6
Python 
import re
import html

def sanitize_email(raw_email):
    clean = re.sub(r'<[^>]+>', ' ', raw_email)  # Remove HTML tags
    clean = html.unescape(clean)                 # Decode HTML entities
    clean = re.sub(r'^[-=_]{3,}$', '', clean, flags=re.MULTILINE)
    clean = re.sub(r'\
{3,}', '\
\
', clean)
    clean = re.sub(r'[\x00-\x08\x0b\x0c\x0e-\x1f\x7f]', '', clean)
    return clean.strip()
Email Sanitizer â€” JavaScript (n8n Code Node) Appendix A, Snippet 7

function sanitizeEmail(rawEmail) {
  return rawEmail
    .replace(/<[^>]+>/g, ' ')         // Remove HTML tags
    .replace(/&amp;/g, '&')           // Decode HTML entities
    .replace(/&lt;/g, '<')
    .replace(/&gt;/g, '>')
    .replace(/&nbsp;/g, ' ')
    .replace(/^[-=_]{3,}$/gm, '')     // Remove separator lines
    .replace(/\
{3,}/g, '\
\
')       // Collapse blank lines
    .trim();
}

const rawEmail = $input.first().json['Text Content'];  // ADJUST FIELD NAME
const cleanEmail = sanitizeEmail(rawEmail);
return [{ json: { clean_email: cleanEmail } }];

Token Cost Calculator â€” Python Appendix A, Snippet 10

import anthropic

client = anthropic.Anthropic(api_key="your-api-key-here")  # CHANGE THIS

response = client.messages.create(
    model="claude-sonnet-4-6",
    max_tokens=500,
    messages=[{"role": "user", "content": "Your prompt here"}]
)

input_tokens = response.usage.input_tokens
output_tokens = response.usage.output_tokens

# Approximate costs (check console.anthropic.com for current pricing)
input_cost = (input_tokens / 1_000_000) * 3.00    # $3 per million input tokens
output_cost = (output_tokens / 1_000_000) * 15.00  # $15 per million output tokens
total_cost = input_cost + output_cost

print(f"Input tokens: {input_tokens}")
print(f"Output tokens: {output_tokens}")
print(f"Estimated cost: ${total_cost:.4f}")
